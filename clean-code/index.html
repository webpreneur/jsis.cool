1.) Prefer using an object as a function parameter, instead of multiple params. 

/** Bad */
function createCar(gps: boolean, radio: boolean) {
    /** ... return a Car ...*/
}

/** Good */
interface ICarExtras {
    gps: boolean;
    radio: boolean;
}
function createCar(extras: ICarExtras) {
    /** ... return a Car ...*/
}
2.) Prefer the Boolean class over the double negation syntax. 

function isValid(obj) {
    return !!obj;
}

function isValid(obj) {
    return Boolean(obj);
}
3.) Do not use magic strings. Keep them in a constant instead for better readibility. 

/** Bad */
function isExportAllowed(exportFileLength: number): boolean {
    return exportFileLength <= 1000;
}
/** Good */
function isExportAllowed(exportFileLength: number): boolean {
    const allowedMaxLength = 1000;
    return exportFileLength <= allowedMaxLength;
}
